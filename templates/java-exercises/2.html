Exercise 1: input [5, 8, 3, 1, 2, 9]

Sắp xếp tăng dần bỏ qua vị trí số chẵn

output [1, 8, 3, 5, 2, 9] 

public class Main {
    public static void main(String[] args) {
        int a[] = {5, 8, 3, 1, 2, 9};
        for (int i = 0; i < a.length - 1; ++i) {
            if (a[i] % 2 == 0) {
                continue;
            }
            for (int j = i + 1; j < a.length; ++j) {
                if (a[j] % 2 == 0) {
                    continue;
                }
                if (a[i] > a[j]) {
                    int tmp = a[i];
                    a[i] = a[j];
                    a[j] = tmp;
                }
            }
        }
        for (int k  = 0; k < a.length; ++k) {
            System.out.print(a[k] + " ");
        }
    }
}
