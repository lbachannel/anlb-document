Exercise 1: Viết hàm kiểm tra chuỗi đối xứng.

Là chuỗi đối xứng => return true

Không phải chuỗi đối xứng => return false 

public class Main {
    public static void main(String[] args) {
        boolean isReverse = checkReverse("level");
        String tmp = isReverse ? "Is Symmetrical Chain" : "Not a Symmetrical Chain";
        System.out.println(tmp);
    }

    public static boolean checkReverse(String value) {
        StringBuilder sb = new StringBuilder(value);
        String reverse = sb.reverse().toString();
        if (value.equalsIgnoreCase(reverse)) {
            return true;
        }
        return false;
    }
}
