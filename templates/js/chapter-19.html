Các cách lấy kích thước của Element

HTML

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <style>
        .box {
            width: 100px;
            height: 100px;
            background: red;   
        }
    </style>
</head>
<body>
    <div class="box"></div>
</body>
</html>

JavaScript

const box = document.querySelector(".box");

// Client: content + padding - scrollbar => integer
console.log("clientWith: ", box.clientWith);
console.log("clientHeight: ", box.clientHeight);

// Offset: content + padding + border + scrollbar => integer
console.log("offsetWidth: ", box.offsetWidth);
console.log("offsetHeight: ", box.offsetHeight);

// Rect: content + padding + border + scrollbar (display) => float
const rect = box.getBoundingClientRect();
console.log("rect.width: ", rect.width);
console.log("rect.height: ", rect.height);
