Generics on Set

public static void main(String[] args) {
    Set<Integer> set = new HashSet();
    set.add(1);
    set.add(2);
    set.add(3);

    int sum = 0;
    Iterator<Integer> it = set.iterator();
    while(it.hasNext()) {
        int element = it.next();
        sum += element;
    }
    System.out.println(sum);
}
