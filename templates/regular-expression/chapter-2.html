1. Kiểm tra một số nằm trong một phạm vi quy định trước, không nhất thiết phải từ 0 đến 9 

public static void main(String[] args) {
    boolean flag;
    Scanner sc = new Scanner(System.in);
    do {
        /*
         *  Số cho phép sử dụng từ 5 đến 9
         *  Tối thiểu 1 chữ số
         *  Tối đa 3 chữ số
        */
        String pattern = "[5-9]{1, 3}";
        System.out.print("Input your house number: ");
        String input = sc.next();
        flag = input.matches(pattern);
        if (!flag) {
            System.out.println("Invalid data");
        }
    } while(!flag);
    System.out.println("Valid data");
}

5. Kiểm tra một chuỗi Không bao gồm các kí tự được quy định trước 

public static void main(String[] args) {
    boolean flag;
    Scanner sc = new Scanner(System.in);
    do {
        /*
            *  Nếu là số thì giá trị được phép sử dụng ngoại trừ 1 3 5 6

            *  Tối thiểu 2 kí tự
            *  Tối đa 8 kí tự
            *  Bắt buộc là chữ hoa
        */

        System.out.print("Input your street name: ");
        String input = sc.next();
        flag = input.matches(pattern);
        if (!flag) {
            System.out.println("Invalid data");
        }
    } while(!flag);
    System.out.println("Valid data");
}
